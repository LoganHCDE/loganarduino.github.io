// This code will light up three LEDs when the button is pressed, and then fade after 2 seconds.

// constant for pin 6, green LED.
const int ledPin1 = 6;


// constant for pin 5, Yellow LED.
const int ledPin2 = 5; 

// constant for pin 9, red LED.
const int ledPin3 = 9; 


// constant for pin 3, for Button.

const int buttonPin = 3;

// The setup() function 

void setup() {

  //  pin 6 setup
  pinMode(ledPin1, OUTPUT);


  //  pin 5 setup
  pinMode(ledPin2, OUTPUT);


  //  pin 9 setup
  pinMode(ledPin3, OUTPUT);


  //  button setup
  pinMode(buttonPin, INPUT_PULLUP); 

}


// The loop() function runs continuously after setup() is complete.

void loop() {

  // see if button is High or Low
  int buttonState = digitalRead(buttonPin);


  // Check if the button is being pressed
  if (buttonState == LOW) {



    // turn on pin 6
    digitalWrite(ledPin1, HIGH);


    // turn on pin 5
    digitalWrite(ledPin2, HIGH);

    // turn on pin 9
    digitalWrite(ledPin3, HIGH);


    // Let 2ds stay on for 2 seconds
    delay(200);


    // for loop for decreasing brightness level
    for (int brightness = 255; brightness >= 0; brightness--) {

      // Set brightness for pin 6
      analogWrite(ledPin1, brightness);


      // Set brightness for pin 5
      analogWrite(ledPin2, brightness);


      // Set brightness for pin 9
      analogWrite(ledPin3, brightness);


      // increase fade time
      delay(10);

    }

  }

}

    Per LED: I = V/R = 5V / 220Ω ≈ 22.7 mA
    All 3 LEDs: Itotal = 22.7 mA × 3 ≈ 68.1 mA
    Runtime: Total Current / Battery Capacity
    Runtime: 68.1 mA / 1200 mAh ≈ 17.6 hours
